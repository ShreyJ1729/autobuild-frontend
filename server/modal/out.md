    setFormState((prevState) => ({ ...prevState, title: newTitle }));

  return (
    <input
      type="text"
      className="rounded-lg p-3 outline-none shadow-md focus:ring-2 ring-blue-200 ring-opacity-50"
      placeholder="Enter title"
      value={title}
      onChange={handleTitleChange}
    />
  );
};

  title: PropTypes.string.isRequired,
  setFormState: PropTypes.func.isRequired
};

import React from "react";
import PropTypes from "prop-types";

type NoteSubmitButtonProps = {
  submitNote: (title: string, message: string, id: number) => boolean,
  setNoteListState: React.Dispatch<React.SetStateAction<{ title: string, message: string, id: number }[]>>
};

const NoteSubmitButton: React.FC<NoteSubmitButtonProps> = ({ submitNote, setNoteListState }) => {
  const handleClick = (): void => {
    const id = new Date().getTime();
    const title = (document.getElementById('note-title') as HTMLInputElement).value;
    const message = (document.getElementById('note-message') as HTMLInputElement).value;
    if (title.trim() === '' || message.trim() === '') {
      alert('Title and message cannot be empty');
    } else {
      submitNote(title, message, id);
      setNoteListState((prevState) => [...prevState, { title, message, id }]);
    }
  };

  return (
    <button type="button" onClick={handleClick} className="bg-green-500 hover:bg-green-700 text-white font-bold py-2 px-4 rounded">
      Add Note
    </button>
  );
};

NoteSubmitButton.propTypes = {
  submitNote: PropTypes.func.isRequired,
  setNoteListState: PropTypes.func.isRequired
};

export default NoteSubmitButton;

import React, { useState } from 'react';
import Header from './Header';
import NoteList from './NoteList';
import NoteForm from './NoteForm';

type Note = {
  title: string;
  message: string;
  id: number;
};

type FormState = {
  title: string;
  message: string;
};

const App = (): JSX.Element => {
  const [noteListState, setNoteListState] = useState<Note[]>([]);
  const [formState, setFormState] = useState<FormState>({ title: '', message: '' });

  const submitNote = (title: string, message: string, id: number): boolean => {
    const newNote = { title, message, id };
    setNoteListState([...noteListState, newNote]);
    setFormState({ title: '', message: '' });
    return true;
  };

  const deleteNote = (id: number): boolean => {
    setNoteListState(noteListState.filter((note) => note.id !== id));
    return true;
  };

  return (
    <div className="h-screen flex flex-col">
      <Header />
      <NoteList noteListState={noteListState} setNoteListState={setNoteListState} />
      <NoteForm formState={formState} setFormState={setFormState} submitNote={submitNote} />
    </div>
  );
};
